Create a TypeScript declaration. Generally, this component shouldn't be used directly, and instead prefer components for specific declarations, e.g. <a href="/reference/typescript/components/enumdeclaration/">EnumDeclaration</a>, <a href="/reference/typescript/components/interfacedeclaration/">InterfaceDeclaration</a>, <a href="/reference/typescript/components/typedeclaration/">TypeDeclaration</a>, etc.

<Tabs syncKey="component-style">
    <TabItem label="jsx">
        <Code code={`import { Declaration } from "@alloy-js/core" 

<Declaration nameKind={TypeScriptElements}>
    {children}
</Declaration>`} lang="tsx" />
</TabItem>
<TabItem label="stc">
    <Code code={`import { Declaration } from "@alloy-js/core/stc" 

Declaration({ nameKind: TypeScriptElements}).children(children)` } lang="ts" />
</TabItem>
</Tabs>

    <td>

  </td>
  </tr>
  <tr>
    <td style="text-align: right;font-weight: bold;">export</td>
    <td>

  </td>
  </tr>
  <tr>
    <td style="text-align: right;font-weight: bold;">kind</td>

  </td>
  </tr>

  </tr>

  </td>
  </tr>
