This component is the root component for all your emitted output. Place your various <a href="/reference/core/components/sourcedirectory/">SourceDirectory</a> and <a href="/reference/core/components/sourcefile/">SourceFile</a> components inside `Output` to create directories and files in your output directory.

<Output

>
    {children}
</Output>`} lang="tsx" />

### Props

<table>
  <tr>
    <td style="text-align: right;font-weight: bold;">basePath</td>
    <td>
      <Badge text="optional" variant="note" size="small" /> <code>string</code>

  </td>
  </tr>
  <tr>
    <td style="text-align: right;font-weight: bold;">children</td>
    <td>
      <Badge text="optional" variant="note" size="small" /> <code><a href="/reference/core/types/children/">Children</a></code>
  
  
  
  
  </td>
  </tr>
  <tr>
    <td style="text-align: right;font-weight: bold;">externals</td>
    <td>
      <Badge text="optional" variant="note" size="small" /> <code><a href="/reference/core/types/symbolcreator/">SymbolCreator</a>[]</code>
  
  External libraries whose symbols should be available for reference.
  
  
  </td>
  </tr>
  <tr>
    <td style="text-align: right;font-weight: bold;">nameConflictResolver</td>
    <td>
      <Badge text="optional" variant="note" size="small" /> <code>(name: string, symbols: any[]) => void</code>
  
  Policy for handling multiple symbols declared with the same name.
  
  
  </td>
  </tr>
  <tr>

    <td>
      <Badge text="optional" variant="note" size="small" /> <code><a href="/reference/core/contexts/namepolicy/">NamePolicy context interface</a>&lt;string></code>

  </td>
  </tr>
</table>
