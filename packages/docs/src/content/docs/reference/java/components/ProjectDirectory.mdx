Represents a java project directory. Use if you want to generate a Java project with a build tool included (maven, gradle etc).

<Tabs syncKey="component-style">
    <TabItem label="jsx">
        <Code code={`import { ProjectDirectory } from "@alloy-js/core" 

<ProjectDirectory 
    gradleProjectConfig={GradleProjectConfig}
    mavenProjectConfig={MavenProjectConfig}
    name="string"
>
    {children}
</ProjectDirectory>`} lang="tsx" />
</TabItem>
<TabItem label="stc">
    <Code code={`import { ProjectDirectory } from "@alloy-js/core/stc" 

ProjectDirectory({ 
    gradleProjectConfig: GradleProjectConfig,
    mavenProjectConfig: MavenProjectConfig,
    name: string
}).children(children)` } lang="ts" />
</TabItem>
</Tabs>
