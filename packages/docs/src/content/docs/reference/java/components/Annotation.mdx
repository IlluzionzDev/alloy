Use an annotation in code, not declaring a new annotation. For instance, use this if you want to annotate a method with 'Override'.

<Tabs syncKey="component-style">
    <TabItem label="jsx">
        <Code code={`import { Annotation } from "@alloy-js/core" 

<Annotation type={Child} value={Record<string, Child>}>
    {children}
</Annotation>`} lang="tsx" />
</TabItem>
<TabItem label="stc">
    <Code code={`import { Annotation } from "@alloy-js/core/stc" 

Annotation({ type: Child, value: Record<string, Child>}).children(children)` } lang="ts" />
</TabItem>
</Tabs>
